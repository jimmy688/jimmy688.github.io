# 遍历目录下的所有文件

遍历目录下的所有文件，并输出文件名：

for file in `ls ./`
do
    echo $file
done

遍历目录下的所有文件，并修改文件名：

#!/bin/bash
  
for file in `ls ./`
do
    result=`echo $file | grep old`
    if [ "$result" != "" ]
    then
        new=${file/old/new}
        mv $file $new
    fi
done
